自动提示框

使用方法


	 <input type="text" id="InputID" value="" >    //声明输入框

	 AutoComplete.create($('#InputID'),[array],options);  //为输入框绑定自动提示功能
	 //or//
	 $('#InputID').autocompleteArray([array],options);     //为输入框绑定自动提示功能
	//$('#InputID')提示框的jquery对象元素
	//[array]：提示单词数组，如['abc', 'bbb','ccc']
	//[options]：提示框的设置选项，具体设置内容见下面设置列表。

设置选项


  options.inputClass = options.inputClass || "ac_input";      //设置input样式，默认为ac_input
  options.resultsClass = options.resultsClass || "ac_results";   //设置提示菜单样式，默认为ac_results
  options.lineSeparator = options.lineSeparator || "\n";    //设置行分割
  options.cellSeparator = options.cellSeparator || "|";       //设置元素分割
  options.minChars = options.minChars || 1;               //设置最小激发提示框的输入字符数，默认为1
  options.delay = options.delay || 400;                     //设置提示框弹出延迟，默认为0.4秒
  options.matchCase = options.matchCase || 0;          //是否区分大小写，默认不区分
  options.matchSubset = options.matchSubset || 1;       //是否匹配所有子集，通常为是，可以提高性能
  options.matchContains = options.matchContains || 0;   //内部子集匹配，如“ba” 可匹配“foot bar” 通常 使用cache时才生效
  options.cacheLength = options.cacheLength || 1;     //cache长度，默认为1，不使用cache，
  options.mustMatch = options.mustMatch || 0;       //必须匹配才可以输入。
  options.extraParams = options.extraParams || {};        //Extra parameters for the backend. If you were to specify { bar:4 }, the autocompleter would call my_autocomplete_backend.php?q=foo&bar=4 (assuming the input box contains "foo").
  options.loadingClass = options.loadingClass || "ac_loading";       //载入时样式         
  options.maxItemsToShow = options.maxItemsToShow || -1;      //最大提示条目数量
  options.autoFill = options.autoFill || false;        //自动填充提示内容
  options.width = parseInt(options.width, 10) || 0;   //提示框宽度
  options.onItemSelect = options.onItemSelect || {}    //当选择一个自动提示时执行的函数。 传入的参数为选中的li元素
